## Level 2

```python
from pathlib import Path
import pandas as pd

def find_data(name):
    for candidate in [
        Path("../../datasets") / name,
        Path("../datasets") / name,
        Path("datasets") / name,
    ]:
        if candidate.exists():
            return candidate
    raise FileNotFoundError(name)

df = pd.read_csv(find_data("scores.csv"))
print(df.head())
print(df.groupby("major")["score"].mean().round(1))
high = df.loc[df["score"] >= 85, ["name", "major", "score"]]
print(high)
assert {"name", "major", "score"}.issubset(df.columns)
```

## Level 3

```python
summary = df["score"].describe()
print(summary)
assert summary["min"] >= 0 and summary["max"] <= 100
```
